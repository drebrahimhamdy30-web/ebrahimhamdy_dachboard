<script>
    async function sendData() {
        const btn = document.getElementById('submitBtn');
        const statusDiv = document.getElementById('status');
        const item = document.getElementById('itemName').value;
        const branch = document.getElementById('branch').value;
        const qty = document.getElementById('quantity').value || 1;

        if (!item) {
            alert("من فضلك اكتب اسم الصنف");
            return;
        }

        // تحضير الواجهة للحالة "جاري الإرسال"
        btn.disabled = true;
        btn.innerText = "جاري الحفظ...";
        statusDiv.style.display = "none";

        try {
            // ضع رابط الـ Webhook الخاص بـ n8n هنا
            const webhookUrl = 'https://YOUR_N8N_URL/webhook-test/shortage-test';

            const response = await fetch(webhookUrl, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    itemName: item,
                    branch: branch,
                    quantity: qty,
                    employee: "Staff", // يمكنك تطويرها لاحقاً لإضافة اسم الموظف
                    date: new Date().toLocaleString('ar-EG')
                })
            });

            if (response.ok) {
                const resData = await response.json();
                statusDiv.className = "success-msg";
                statusDiv.innerText = resData.message || "تم تسجيل الناقص بنجاح ✅";
                document.getElementById('itemName').value = ""; // مسح الخانة بعد النجاح
            } else {
                throw new Error();
            }
        } catch (error) {
            statusDiv.className = "error-msg";
            statusDiv.innerText = "حدث خطأ في الاتصال بالسيرفر ❌";
        } finally {
            statusDiv.style.display = "block";
            btn.disabled = false;
            btn.innerText = "تسجيل الطلب";
        }
    }
</script>
