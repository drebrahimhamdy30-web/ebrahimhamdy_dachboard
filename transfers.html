async function loadTransfers() { const tbody = document.getElementById('table-body'); tbody.innerHTML = 'جاري جلب التحويلات...'; const response = await fetchData(); // الوصول الصحيح للبيانات بناءً على الصورة التي أرفقتها if (response && response[0] && response[0].data) { rawData = response[0].data; // الفلترة مع دعم المسميات العربية (تحويل) والإنجليزية (transfer) const filteredData = rawData.filter(item => { const isTransfer = (item.type === 'transfer' || item.type === 'تحويل'); // التحقق من أن المستخدم الحالي هو الطالب أو هو الفرع المطلوب منه const involvesMe = (item.user === activeUser || item.branch === userBranch || item.branch === activeUser); return isTransfer && involvesMe; }); if (filteredData.length > 0) { renderTable(filteredData); } else { tbody.innerHTML = 'لا توجد تحويلات تخص فرعك حالياً'; } } else { tbody.innerHTML = 'تعذر جلب البيانات من السيرفر'; } } function renderTable(items) { const tbody = document.getElementById('table-body'); tbody.innerHTML = ''; items.forEach((order, index) => { const row = document.createElement('tr'); const isTargetBranch = (order.branch === userBranch || order.branch === activeUser); const isDisabled = isTargetBranch ? '' : 'disabled'; const currentState = order.state || 'pending'; const note = order.company || ''; // حل مشكلة الـ undefined: إذا لم يوجد id نستخدم رقم الصف (index + 1) const displayId = order.id || (index + 1); row.innerHTML = ` ${displayId} ${order.user || '-'} ${order.branch || '-'} ${order.item || '-'} ${order.qty || '0'} ${isTargetBranch ? `` : 'بانتظار الرد'} `; tbody.appendChild(row); }); }
